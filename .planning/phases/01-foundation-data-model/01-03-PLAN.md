import { db } from '../db'
import { categories } from '../db/schema'
import { count } from 'drizzle-orm'

  const [result] = await db.select({ count: count() }).from(categories)

    timestamp: new Date().toISOString(),

})

export default health

import { Hono } from 'hono'
import { db } from '../db'
import { categories } from '../db/schema'
import { eq } from 'drizzle-orm'

const categoriesRoute = new Hono()

categoriesRoute.get('/', async (c) => {
  const result = await db
    .select()
    .from(categories)
    .where(eq(categories.active, true))
    .orderBy(categories.sortOrder)

  return c.json(result)
})

export default categoriesRoute

Update server/src/index.ts to:
1. Import and mount health and categories routes
2. Run migrations on startup (ensures DB exists when server starts)
3. Run seed on startup (idempotent — safe to run every time)

```typescript
import { serve } from '@hono/node-server'
import { Hono } from 'hono'
import { cors } from 'hono/cors'
import health from './routes/health'
import categoriesRoute from './routes/categories'

// Ensure DB is migrated and seeded on startup
import './db/migrate-and-seed'

const app = new Hono()

app.use('/api/*', cors())
app.route('/api/health', health)
app.route('/api/categories', categoriesRoute)

const port = 3001
console.log(`Server running on http://localhost:${port}`)
serve({ fetch: app.fetch, port })

export type AppType = typeof app
```

Create server/src/db/migrate-and-seed.ts that runs migrations then seed (so server is always ready):
```typescript
import { migrate } from 'drizzle-orm/better-sqlite3/migrator'
import { db } from './index'
import { categories } from './schema'
import { SEED_CATEGORIES } from './seed-data'

// Run migrations
migrate(db, { migrationsFolder: './src/db/migrations' })

// Seed categories (idempotent)
for (const cat of SEED_CATEGORIES) {
  db.insert(categories).values(cat).onConflictDoNothing({ target: categories.name }).run()
}
```

Extract SEED_CATEGORIES from seed.ts into a shared seed-data.ts file that both seed.ts and migrate-and-seed.ts can import.
